import java.util.Random;

//این کلاس همانند کلاس EnemyShip عمل میکند :
public class SpaceDust {

    private int x, y;
    private int speed;

    // Detect dust leaving the screen
    private int maxX;
    private int maxY;
    private int minX;
    private int minY;

    // Constructor
//در سازنده ی کلاس مقادیری که برای نمایش اولیه ی گرد و غبار لازم است تعیین میشود :
    public SpaceDust(int screenX, int screenY){

        maxX = screenX;
        maxY = screenY;
        minX = 0;
        minY = 0;

        Random generator = new Random();
        speed = generator.nextInt(10);

        x = generator.nextInt(maxX);
        y = generator.nextInt(maxY);
    }

//این متد همانند توضیحات متد update از کلاس EnemyShip عمل میکند و مکان جدید در حین بازی را برای نمایش گرد و غبار تعیین میکند :
    public void update(int playerSpeed){
        x -= playerSpeed;
        x -= speed;

        //respawn space dust
        if(x < 0){
            x = maxX;
            Random generator = new Random();
            y = generator.nextInt(maxY);
            speed = generator.nextInt(15);
        }
    }

    //Getters and Setters
    public int getX() {

        return x;
    }

    public int getY() {

        return y;
    }
}
